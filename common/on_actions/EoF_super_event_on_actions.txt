on_actions = {
	on_declare_war = { #Celestial Sea War
		effect = {
			if = {
				limit = {
					tag = JER
					FROM = { tag = EQS }
				}
				every_country = {
					set_country_flag = EoF_big_event_custom_island_war
				}
			}
		}
	}
	on_declare_war = { #Great War
		effect = {
			if = {
				limit = {
					tag = CHN
					FROM = { tag = EQS }
				}
				every_country = {
					set_country_flag = EoF_big_event_custom_great_war
				}
			}
		}
	}
	on_state_control_changed = { #Sunset's Counteroffensive
		effect = {
			if = {
				limit = {
					NOT = {
						has_global_flag = EoF_sunset_event_fired
					}
					EQS = {
						surrender_progress > 0.2
					}
					FROM = {
						OR = {
							is_in_faction_with = EQS
							tag = EQS
						}
					}
					ROOT = {
						OR = {
							is_in_faction_with = CHN
							tag = CHN
						}
					}
					FROM.FROM = {
						OR = {
							state = 66
							state = 119
							state = 114
							state = 101
							state = 68
							state = 179
							state = 178
							state = 70
							state = 148
							state = 147
							state = 144
							state = 141
						}
					}
				}
				every_country = {
					set_country_flag = EoF_big_event_custom_sunset
				}
				set_global_flag = EoF_sunset_event_fired
			}
		}
	}
}
